  <script>
    function showToast(msg, type) {
      const t = document.getElementById('toast');
      t.textContent = msg;
      t.className = 'toast toast-' + type;
      setTimeout(() => t.className = 'toast', 4000);
    }

    async function cancelTicket(id) {
      if (!confirm('Cancel this ticket? It will no longer be valid for entry.')) return;
      try {
        const res = await fetch('/admin/api/ticket/' + id + '/cancel', { method: 'POST', headers: { 'Content-Type': 'application/json' } });
        const data = await res.json();
        if (data.success) { showToast('Ticket cancelled.', 'success'); setTimeout(() => location.reload(), 1000); }
        else { showToast(data.message, 'error'); }
      } catch (e) { showToast('Error.', 'error'); }
    }

    async function reissueTicket(id) {
      if (!confirm('Reissue this ticket? A new ticket number and QR code will be generated.')) return;
      try {
        const res = await fetch('/admin/api/ticket/' + id + '/reissue', { method: 'POST', headers: { 'Content-Type': 'application/json' } });
        const data = await res.json();
        if (data.success) { showToast('Ticket reissued.', 'success'); setTimeout(() => location.reload(), 1000); }
        else { showToast(data.message, 'error'); }
      } catch (e) { showToast('Error.', 'error'); }
    }

    async function showQR(id) {
      try {
        const res = await fetch('/api/ticket/' + id + '/qr');
        const data = await res.json();
        if (data.success) {
          const win = window.open('', '_blank');
          win.document.write(`
            <html><head><title>Ticket QR Code</title>
            <style>
              body { display:flex;align-items:center;justify-content:center;min-height:100vh;background:#0A0A0A;flex-direction:column;font-family:sans-serif; }
              img { max-width:300px;border-radius:12px; }
              p { color:#888;margin-top:12px;font-size:13px; }
            </style></head>
            <body>
              <img src="${data.qrImage}" alt="QR Code" />
              <p>Ticket: ${data.ticketNumber}</p>
              <p style="font-size:11px;color:#555;">Scan this QR at the venue for entry</p>
            </body></html>
          `);
        } else { showToast('Error loading QR.', 'error'); }
      } catch (e) { showToast('Error.', 'error'); }
    }
  </script>
